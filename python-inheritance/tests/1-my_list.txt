import unittest
from 1_my_list import MyList  # Cambia esto por el nombre correcto del archivo

class TestMyList(unittest.TestCase):
    
    def test_print_sorted(self):
        # Captura la salida de print
        import io
        import sys
        
        # Crea una instancia de MyList
        my_list = MyList([3, 1, 2])
        
        # Redirige la salida estándar
        captured_output = io.StringIO()
        sys.stdout = captured_output
        
        # Llama al método print_sorted
        my_list.print_sorted()
        
        # Restaura la salida estándar
        sys.stdout = sys.__stdout__
        
        # Verifica que la salida sea la esperada
        self.assertEqual(captured_output.getvalue().strip(), "[1, 2, 3]")
        
    def test_empty_list(self):
        # Prueba con una lista vacía
        my_list = MyList([])
        captured_output = io.StringIO()
        sys.stdout = captured_output
        my_list.print_sorted()
        sys.stdout = sys.__stdout__
        self.assertEqual(captured_output.getvalue().strip(), "[]")

    def test_negative_numbers(self):
        # Prueba con números negativos
        my_list = MyList([-1, -3, -2])
        captured_output = io.StringIO()
        sys.stdout = captured_output
        my_list.print_sorted()
        sys.stdout = sys.__stdout__
        self.assertEqual(captured_output.getvalue().strip(), "[-3, -2, -1]")
        
    def test_mixed_numbers(self):
        # Prueba con números mixtos (positivos y negativos)
        my_list = MyList([3, -1, 2, -3])
        captured_output = io.StringIO()
        sys.stdout = captured_output
        my_list.print_sorted()
        sys.stdout = sys.__stdout__
        self.assertEqual(captured_output.getvalue().strip(), "[-3, -1, 2, 3]")

if __name__ == '__main__':
    unittest.main()
